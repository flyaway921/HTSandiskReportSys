CREATE TABLE [_group] (
  [id] INTEGER NOT NULL ON CONFLICT IGNORE PRIMARY KEY ON CONFLICT IGNORE AUTOINCREMENT, 
  [name] CHAR(50) NOT NULL ON CONFLICT IGNORE, 
  [members] BLOB);
  
CREATE TABLE [user] (
  [id] INTEGER NOT NULL ON CONFLICT IGNORE PRIMARY KEY ON CONFLICT IGNORE AUTOINCREMENT, 
  [username] CHAR(50) NOT NULL ON CONFLICT IGNORE, 
  [pwd] CHAR(50) NOT NULL ON CONFLICT IGNORE, 
  [mail] CHAR(50) NOT NULL ON CONFLICT IGNORE);
  
CREATE TABLE report(
  id INTEGER NOT NULL ON CONFLICT IGNORE PRIMARY KEY ON CONFLICT IGNORE AUTOINCREMENT,
  author_id INTEGER NOT NULL,
  content TEXT);
